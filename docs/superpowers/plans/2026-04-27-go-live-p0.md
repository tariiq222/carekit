# Go-Live P0 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close the five P0 launch blockers identified in the 2026-04-27 Go-Live audit: per-tenant Moyasar credentials, ZATCA UBL 2.1 + QR + hash-chain infrastructure, real refund-to-Moyasar wiring with credit-notes, recurring subscription charging, and public Terms/Privacy/DPA pages.

**Architecture:**
- **TWO Moyasars** — `MOYASAR_PLATFORM_*` envs stay (CareKit collecting from tenants); a new `OrganizationPaymentConfig` singleton stores each tenant's own Moyasar credentials encrypted via AES-GCM with `organizationId` as AAD (mirrors `SmsCredentialsService`).
- **ZATCA disabled-by-default** — full UBL 2.1 builder + Phase-2 TLV QR + previous-invoice hash chain implemented behind `ZATCA_ENABLED=false`; onboarding documented in a runbook so a future tenant flips one flag.
- **Refund flow** — `RefundPaymentHandler` calls `MoyasarApiClient.refund` with the *tenant* key, creates a `CreditNote` row, and (if ZATCA enabled) auto-submits the credit-note.
- **Recurring charging** — `ChargeDueSubscriptionsCron` finishes the existing TODO by calling `MoyasarSubscriptionClient.chargeWithToken` against the platform key.
- **Legal pages** — three static AR/EN pages on `apps/website` rendered from `messages/{ar,en}.json` strings; linked from footer.

**Tech Stack:** NestJS 11, Prisma 7 (split schemas), AES-256-GCM (`crypto`), `qrcode` (Phase-2 TLV), `xmlbuilder2` (UBL 2.1), Next.js 15 App Router, next-intl.

`apps/backend/src/infrastructure/payments/moyasar-credentials.service.ts`:

```typescript
// AES-256-GCM credential store for per-tenant Moyasar keys.
// Mirror of SmsCredentialsService — organizationId is bound as AAD so a
// ciphertext from one tenant cannot be replayed against another.

import { Injectable, InternalServerErrorException } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { createCipheriv, createDecipheriv, randomBytes } from 'crypto';

@Injectable()
export class MoyasarCredentialsService {
  private readonly key: Buffer;

  constructor(cfg: ConfigService) {
    const raw = cfg.get<string>('MOYASAR_TENANT_ENCRYPTION_KEY');
    if (!raw) {
      throw new InternalServerErrorException('MOYASAR_TENANT_ENCRYPTION_KEY missing');
    }
    const buf = Buffer.from(raw, 'base64');
    if (buf.length !== 32) {
      throw new InternalServerErrorException('MOYASAR_TENANT_ENCRYPTION_KEY must decode to 32 bytes');
    }
    this.key = buf;
  }

  encrypt(payload: Record<string, unknown>, organizationId: string): string {
    const iv = randomBytes(12);
    const cipher = createCipheriv('aes-256-gcm', this.key, iv);
    cipher.setAAD(Buffer.from(organizationId, 'utf8'));
    const plain = Buffer.from(JSON.stringify(payload), 'utf8');
    const ct = Buffer.concat([cipher.update(plain), cipher.final()]);
    const tag = cipher.getAuthTag();
    return Buffer.concat([iv, tag, ct]).toString('base64');
  }

  decrypt<T extends Record<string, unknown>>(ciphertext: string, organizationId: string): T {
    const buf = Buffer.from(ciphertext, 'base64');
    const iv = buf.subarray(0, 12);
    const tag = buf.subarray(12, 28);
    const ct = buf.subarray(28);
    const decipher = createDecipheriv('aes-256-gcm', this.key, iv);
    decipher.setAAD(Buffer.from(organizationId, 'utf8'));
    decipher.setAuthTag(tag);
    const plain = Buffer.concat([decipher.update(ct), decipher.final()]);
    return JSON.parse(plain.toString('utf8')) as T;
  }
}
```

// apps/backend/src/modules/finance/moyasar-config/upsert-moyasar-config.handler.ts
import { Injectable } from '@nestjs/common';
import { PrismaService } from '../../../infrastructure/database';
import { TenantContextService } from '../../../common/tenant';
import { MoyasarCredentialsService } from '../../../infrastructure/payments/moyasar-credentials.service';

export interface UpsertMoyasarConfigCommand {
  publishableKey: string;
  secretKey: string;
  webhookSecret?: string;
  isLive: boolean;
}

@Injectable()
export class UpsertMoyasarConfigHandler {
  constructor(
    private readonly prisma: PrismaService,
    private readonly creds: MoyasarCredentialsService,
    private readonly tenant: TenantContextService,
  ) {}

  async execute(cmd: UpsertMoyasarConfigCommand) {
    const organizationId = this.tenant.requireOrganizationIdOrDefault();
    const secretKeyEnc = this.creds.encrypt({ secretKey: cmd.secretKey }, organizationId);
    const webhookSecretEnc = cmd.webhookSecret
      ? this.creds.encrypt({ webhookSecret: cmd.webhookSecret }, organizationId)
      : null;
    return this.prisma.organizationPaymentConfig.upsert({
      where: { organizationId },
      create: { organizationId, publishableKey: cmd.publishableKey, secretKeyEnc, webhookSecretEnc, isLive: cmd.isLive },
      update: { publishableKey: cmd.publishableKey, secretKeyEnc, webhookSecretEnc, isLive: cmd.isLive },
    });
  }
}

// apps/backend/src/modules/finance/moyasar-config/test-moyasar-config.handler.ts
import { Injectable, BadRequestException } from '@nestjs/common';
import { PrismaService } from '../../../infrastructure/database';
import { TenantContextService } from '../../../common/tenant';
import { MoyasarCredentialsService } from '../../../infrastructure/payments/moyasar-credentials.service';

@Injectable()
export class TestMoyasarConfigHandler {
  constructor(
    private readonly prisma: PrismaService,
    private readonly creds: MoyasarCredentialsService,
    private readonly tenant: TenantContextService,
  ) {}

  async execute(): Promise<{ ok: boolean; status: string }> {
    const organizationId = this.tenant.requireOrganizationIdOrDefault();
    const cfg = await this.prisma.organizationPaymentConfig.findUnique({
      where: { organizationId },
    });
    if (!cfg) throw new BadRequestException('Moyasar not configured for this organization');

    const { secretKey } = this.creds.decrypt<{ secretKey: string }>(cfg.secretKeyEnc, organizationId);

    let status: string;
    let ok = false;
    try {
      const res = await fetch('https://api.moyasar.com/v1/payments?per=1', {
        headers: { Authorization: `Bearer ${secretKey}` },
      });
      status = res.status === 200 ? 'OK' : res.status === 401 ? 'INVALID_KEY' : `HTTP_${res.status}`;
      ok = res.status === 200;
    } catch {
      status = 'NETWORK_ERROR';
    }

    await this.prisma.organizationPaymentConfig.update({
      where: { organizationId },
      data: { lastVerifiedAt: new Date(), lastVerifiedStatus: status },
    });
    return { ok, status };
  }
}

// apps/backend/src/modules/finance/zatca-submit/ubl-builder.ts
// ZATCA Phase-2 simplified/standard tax invoice builder.
// Output: UBL 2.1 XML with PIH (previous invoice hash) for chaining.
// Reference: ZATCA E-Invoicing Implementation Standards (rev 1.6, 2024).

import { create } from 'xmlbuilder2';

export interface UblInvoiceInput {
  invoiceId: string;
  issueDateTime: Date;
  invoiceCounter: number;
  previousInvoiceHash: string; // empty for first invoice
  type: 'standard' | 'simplified';
  seller: {
    vatNumber: string;
    name: string;
    address: { street: string; city: string; postalCode: string; country: string };
  };
  buyer: { name: string; vatNumber?: string };
  lines: Array<{ id: string; name: string; quantity: number; unitPrice: number; taxRate: number }>;
  totals: { subtotal: number; vat: number; total: number; currency: string };
}

export function buildUblInvoiceXml(input: UblInvoiceInput): string {
  const root = create({ version: '1.0', encoding: 'UTF-8' })
    .ele('Invoice', {
      xmlns: 'urn:oasis:names:specification:ubl:schema:xsd:Invoice-2',
      'xmlns:cac': 'urn:oasis:names:specification:ubl:schema:xsd:CommonAggregateComponents-2',
      'xmlns:cbc': 'urn:oasis:names:specification:ubl:schema:xsd:CommonBasicComponents-2',
      'xmlns:ext': 'urn:oasis:names:specification:ubl:schema:xsd:CommonExtensionComponents-2',
    });
  root.ele('cbc:ProfileID').txt('reporting:1.0');
  root.ele('cbc:ID').txt(input.invoiceId);
  root.ele('cbc:UUID').txt(input.invoiceId);
  root.ele('cbc:IssueDate').txt(input.issueDateTime.toISOString().slice(0, 10));
  root.ele('cbc:IssueTime').txt(input.issueDateTime.toISOString().slice(11, 19));
  root.ele('cbc:InvoiceTypeCode', { name: input.type === 'standard' ? '0100000' : '0200000' }).txt('388');
  root.ele('cbc:DocumentCurrencyCode').txt(input.totals.currency);
  root.ele('cbc:TaxCurrencyCode').txt('SAR');

  // ICV (Invoice Counter Value)
  const icv = root.ele('cac:AdditionalDocumentReference');
  icv.ele('cbc:ID').txt('ICV');
  icv.ele('cbc:UUID').txt(String(input.invoiceCounter));

  // PIH (Previous Invoice Hash)
  const pih = root.ele('cac:AdditionalDocumentReference');
  pih.ele('cbc:ID').txt('PIH');
  const pihAttachment = pih.ele('cac:Attachment');
  const pihEDB = pihAttachment.ele('cbc:EmbeddedDocumentBinaryObject', { mimeCode: 'text/plain' });
  pihEDB.txt(input.previousInvoiceHash);

  // Seller
  const supplier = root.ele('cac:AccountingSupplierParty').ele('cac:Party');
  supplier.ele('cac:PartyIdentification').ele('cbc:ID', { schemeID: 'CRN' }).txt('1010101010');
  const sellerAddr = supplier.ele('cac:PostalAddress');
  sellerAddr.ele('cbc:StreetName').txt(input.seller.address.street);
  sellerAddr.ele('cbc:CityName').txt(input.seller.address.city);
  sellerAddr.ele('cbc:PostalZone').txt(input.seller.address.postalCode);
  sellerAddr.ele('cac:Country').ele('cbc:IdentificationCode').txt(input.seller.address.country);
  const sellerTax = supplier.ele('cac:PartyTaxScheme');
  sellerTax.ele('cbc:CompanyID').txt(input.seller.vatNumber);
  sellerTax.ele('cac:TaxScheme').ele('cbc:ID').txt('VAT');
  supplier.ele('cac:PartyLegalEntity').ele('cbc:RegistrationName').txt(input.seller.name);

  // Buyer
  const customer = root.ele('cac:AccountingCustomerParty').ele('cac:Party');
  customer.ele('cac:PartyLegalEntity').ele('cbc:RegistrationName').txt(input.buyer.name);

  // Tax totals
  const taxTotal = root.ele('cac:TaxTotal');
  taxTotal.ele('cbc:TaxAmount', { currencyID: 'SAR' }).txt(input.totals.vat.toFixed(2));

  // Legal monetary totals
  const lmt = root.ele('cac:LegalMonetaryTotal');
  lmt.ele('cbc:LineExtensionAmount', { currencyID: input.totals.currency }).txt(input.totals.subtotal.toFixed(2));
  lmt.ele('cbc:TaxExclusiveAmount', { currencyID: input.totals.currency }).txt(input.totals.subtotal.toFixed(2));
  lmt.ele('cbc:TaxInclusiveAmount', { currencyID: input.totals.currency }).txt(input.totals.total.toFixed(2));
  lmt.ele('cbc:PayableAmount', { currencyID: input.totals.currency }).txt(input.totals.total.toFixed(2));

  // Lines
  for (const line of input.lines) {
    const il = root.ele('cac:InvoiceLine');
    il.ele('cbc:ID').txt(line.id);
    il.ele('cbc:InvoicedQuantity', { unitCode: 'PCE' }).txt(String(line.quantity));
    il.ele('cbc:LineExtensionAmount', { currencyID: input.totals.currency }).txt((line.quantity * line.unitPrice).toFixed(2));
    il.ele('cac:Item').ele('cbc:Name').txt(line.name);
    il.ele('cac:Price').ele('cbc:PriceAmount', { currencyID: input.totals.currency }).txt(line.unitPrice.toFixed(2));
  }

  return root.end({ prettyPrint: false });
}
```

// apps/backend/src/modules/finance/zatca-submit/qr-tlv.ts
// ZATCA Phase-2 QR — 5 mandatory TLV tags. Reference: ZATCA E-Invoicing
// Implementation Standards §3.3 (rev 1.6).

export interface QrInput {
  sellerName: string;
  sellerVat: string;
  timestamp: Date;
  total: string; // VAT-inclusive total, 2 decimals
  vat: string;   // VAT amount, 2 decimals
}

export function buildPhase2QrTlv(input: QrInput): string {
  const parts: Buffer[] = [
    tlv(0x01, input.sellerName),
    tlv(0x02, input.sellerVat),
    tlv(0x03, input.timestamp.toISOString()),
    tlv(0x04, input.total),
    tlv(0x05, input.vat),
  ];
  return Buffer.concat(parts).toString('base64');
}

function tlv(tag: number, value: string): Buffer {
  const v = Buffer.from(value, 'utf8');
  if (v.length > 0xff) {
    throw new Error(`TLV value too long (${v.length} bytes) for tag ${tag} — ZATCA Phase-2 expects single-byte length`);
  }
  return Buffer.concat([Buffer.from([tag, v.length]), v]);
}
```

```typescript
// refund-payment.handler.ts
import { Injectable, NotFoundException, BadRequestException, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { PaymentStatus, PaymentMethod } from '@prisma/client';
import { PrismaService } from '../../../infrastructure/database';
import { MoyasarApiClient } from '../moyasar-api/moyasar-api.client';
import { ZatcaSubmitHandler } from '../zatca-submit/zatca-submit.handler';

interface RefundPaymentCommand { paymentId: string; reason: string; amount?: number; }

@Injectable()
export class RefundPaymentHandler {
  private readonly logger = new Logger(RefundPaymentHandler.name);
  constructor(
    private readonly prisma: PrismaService,
    private readonly moyasar: MoyasarApiClient,
    private readonly zatca: ZatcaSubmitHandler,
    private readonly config: ConfigService,
  ) {}

  async execute(cmd: RefundPaymentCommand) {
    const payment = await this.prisma.payment.findFirst({ where: { id: cmd.paymentId } });
    if (!payment) throw new NotFoundException('Payment not found');
    if (payment.status !== PaymentStatus.COMPLETED) {
      throw new BadRequestException('Only completed payments can be refunded');
    }
    const refundAmount = cmd.amount ?? Number(payment.amount);
    if (refundAmount > Number(payment.amount)) {
      throw new BadRequestException('Refund amount exceeds payment amount');
    }

    if (payment.method === PaymentMethod.ONLINE_CARD) {
      if (!payment.gatewayRef) throw new BadRequestException('Online payment is missing gateway reference');
      const halalas = Math.round(refundAmount * 100);
      await this.moyasar.refundPayment(payment.organizationId, payment.gatewayRef, halalas);
    }

    const creditNote = await this.prisma.$transaction(async (tx) => {
      await tx.payment.update({
        where: { id: payment.id },
        data: { status: PaymentStatus.REFUNDED, failureReason: cmd.reason },
      });
      return tx.creditNote.create({
        data: {
          organizationId: payment.organizationId,
          invoiceId: payment.invoiceId,
          paymentId: payment.id,
          amount: refundAmount,
          reason: cmd.reason,
          status: 'ISSUED',
        },
      });
    });

    if (this.config.get<string>('ZATCA_ENABLED') === 'true') {
      try {
        const sub = await this.zatca.execute({ invoiceId: payment.invoiceId, type: 'credit-note', creditNoteId: creditNote.id });
        await this.prisma.creditNote.update({ where: { id: creditNote.id }, data: { status: 'ZATCA_SUBMITTED', zatcaSubmissionId: sub.id } });
      } catch (err) {
        this.logger.error(`ZATCA submission failed for credit-note ${creditNote.id}`, err);
      }
    }

    return { paymentId: payment.id, creditNoteId: creditNote.id, refundedAmount: refundAmount };
  }
}
```
